+import stripe
+from flask import current_app, g
+
+from api.core.validation.exceptions import BusinessRuleError
+
+
+def create_mentorship_checkout() -> dict:
+    price_id = current_app.config.get('STRIPE_MENTORSHIP_PRICE_ID')
+    if not price_id:
+        raise BusinessRuleError("Mentorship pricing not configured")
+
+    frontend_url = current_app.config.get('FRONTEND_URL', 'http://localhost:3000')
+    if not frontend_url.endswith('/'):
+        frontend_url += '/'
+
+    application_id = g.validated.get('application_id', '')
+    email = g.validated.get('email', '')
+
+    try:
+        session_params: dict = {
+            "payment_method_types": ["card"],
+            "line_items": [{"price": price_id, "quantity": 1}],
+            "mode": "payment",
+            "success_url": f"{frontend_url}mentorship?payment=success",
+            "cancel_url": f"{frontend_url}mentorship?payment=cancelled",
+            "metadata": {
+                "type": "mentorship_sprint",
+                "application_id": application_id,
+            },
+        }
+
+        if email:
+            session_params["customer_email"] = str(email)
+
+        checkout_session = stripe.checkout.Session.create(**session_params)
+
+        return {
+            "url": checkout_session.url,
+            "session_id": checkout_session.id,
+        }
+
+    except stripe.StripeError as e:
+        current_app.logger.error(
+            f"Stripe error creating mentorship checkout: {str(e)}"
+        )
+        raise BusinessRuleError(
+            f"Failed to create checkout session: {str(e)}"
+        ) from e
