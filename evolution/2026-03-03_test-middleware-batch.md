+from telehook.middleware.batch import Batch

+class TestBatch:
+    async def test_flushes_after_window(self):
+        sent: list[Message] = []
+
+        async def terminal(msg: Message) -> None:
+            sent.append(msg)
+
+        batch = Batch(window_seconds=0.1, max_size=10)
+        await batch(Message(text="first"), terminal)
+        await batch(Message(text="second"), terminal)
+        assert len(sent) == 0
+
+        await asyncio.sleep(0.2)
+        await batch.flush()
+        assert len(sent) == 1
+        assert "first" in sent[0].text
+        assert "second" in sent[0].text
+
+    async def test_flushes_at_max_size(self):
