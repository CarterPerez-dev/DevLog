+def main():
+    notifier = Notifier()
+
+    try:
+        result = do_work()
+        notifier.send_sync(
+            alert_formatter(
+                title="Backup completed",
+                app="backup-script",
+                details=f"Backed up {result} files",
+            )
+        )
+    except Exception as e:
+        notifier.send_sync(
+            error_formatter(
+                exception=e,
+                component="backup",
+                app="backup-script",
+            )
+        )
+        sys.exit(1)
