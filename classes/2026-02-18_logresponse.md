class LogResponse(BaseResponseSchema):
    """
    Schema for log API response
    """
    challenge_id: UUID
    log_date: date
    day_number: int
    tiktok: int
    instagram_reels: int
    youtube_shorts: int
    twitter: int
    reddit: int
    linkedin_personal: int
    linkedin_company: int
    youtube_full: int
    medium: int
    jobs_applied: int

    @computed_field
    @property
    def total_content(self) -> int:
        """
        Calculate total content for this day
        """
        return (
            self.tiktok +
            self.instagram_reels +
            self.youtube_shorts +
            self.twitter +
            self.reddit +
            self.linkedin_personal +
            self.linkedin_company +
            self.youtube_full +
            self.medium
        )
