class LevelService:
    """
    Service class for level and role operations
    """
    @staticmethod
    def get_current_role(level: int) -> str:
        """
        Get the current role name based on user level
        """
        clamped_level = max(1, min(100, level))
        return Business.LEVEL_ROLES.get(clamped_level, "Script-kiddie I")

    @staticmethod
    def get_next_role(level: int) -> dict[str, str | int] | None:
        """
        Get the next role the user will reach
        """
        if level >= 100:
            return None

        next_level = level + 1
        next_role = Business.LEVEL_ROLES.get(next_level, "Script-kiddie I")

        return {"level": next_level, "role": next_role, "levels_to_go": 1}
