function DailyOps({
  dailyBonus,
  practiceRemaining,
  subscriptionActive,
  onClaimDaily,
  isClaimingDaily,
}: DailyOpsProps): React.ReactElement {
  const { isSimple } = useTheme();
  const s = isSimple ? simpleStyles : defaultStyles;
  const { openModal } = useSubscriptionModalStore();
  const formatTimeRemaining = (milliseconds: number): string => {
    const hours = Math.floor(milliseconds / (1000 * 60 * 60));
    const minutes = Math.floor(
      (milliseconds % (1000 * 60 * 60)) / (1000 * 60),
    );

    if (hours > 0) {
      return `${String(hours)}h ${String(minutes)}m remaining`;
    }
    if (minutes > 0) {
      return `${String(minutes)}m remaining`;
    }
    return 'Available now';
  };

  const getDailyBonusStatus = (): {
    status: DailyOpStatus;
    icon: React.ReactElement;
  } => {
    if (dailyBonus.canClaim) {
      return { status: DailyOpStatus.AVAILABLE, icon: <GiHeartburn /> };
    }
    return { status: DailyOpStatus.CLAIMED, icon: <GiPiercedHeart /> };
  };

  const getDailyQuestionStatus = (): {
    status: DailyOpStatus;
    icon: React.ReactElement;
  } => {
    if (!dailyBonus.subscriptionActive) {
      return { status: DailyOpStatus.LOCKED, icon: <GiLockedHeart /> };
    }
    if (dailyBonus.alreadyAnswered) {
      return { status: DailyOpStatus.COMPLETED, icon: <GiOuroboros /> };
    }
    return { status: DailyOpStatus.AVAILABLE, icon: <GiRattlesnake /> };
  };

  const getPracticeStatus = (): DailyOpStatus => {
    if (subscriptionActive) return DailyOpStatus.UNLIMITED;
    if (practiceRemaining > 0) return DailyOpStatus.REMAINING;
    return DailyOpStatus.LOCKED;
  };

  const { status: bonusStatus, icon: bonusIcon } = getDailyBonusStatus();
  const { status: questionStatus, icon: questionIcon } =
    getDailyQuestionStatus();
  const practiceStatus = getPracticeStatus();

  return (
    <div className={s.dailyOps}>
      <div className={s.header}>
        <GiBulletBill className={s.icon} />
        <span>DAILY OPERATIONS</span>
      </div>

      <div className={s.opsGrid}>
        <div
          className={`${s.opCard} ${s[bonusStatus.toLowerCase()]}`}
        >
          <div className={s.opHeader}>
            {bonusIcon}
            <span>DAILY BONUS</span>
          </div>

          <div className={s.opContent}>
            <div className={s.opStatus}>STATUS: {bonusStatus}</div>
            {bonusStatus === DailyOpStatus.AVAILABLE ? (
              <button
                className={s.claimButton}
                onClick={onClaimDaily}
                disabled={isClaimingDaily}
              >
                {isClaimingDaily ? 'CLAIMING...' : 'CLAIM BONUS'}
              </button>
            ) : null}
            {bonusStatus === DailyOpStatus.CLAIMED ? (
              <div className={s.completedMessage}>
                {dailyBonus.timeUntilNextClaim > 0
                  ? formatTimeRemaining(dailyBonus.timeUntilNextClaim)
                  : 'CLAIMED TODAY'}
