async function loadMixamoAnimation(
	url: string,
	vrm: VRM,
): Promise<THREE.AnimationClip> {
	const loader = new FBXLoader();
	const asset = await loader.loadAsync(url);

	const clip = THREE.AnimationClip.findByName(asset.animations, "mixamo.com");
	if (!clip) throw new Error("mixamo.com animation clip not found");

	const tracks: THREE.KeyframeTrack[] = [];

	const restRotationInverse = new THREE.Quaternion();
	const parentRestWorldRotation = new THREE.Quaternion();
	const _quatA = new THREE.Quaternion();

	const motionHipsHeight =
		asset.getObjectByName("mixamorigHips")?.position.y ?? 1;
	const vrmHipsY = vrm.humanoid?.normalizedRestPose?.hips?.position?.[1] ?? 1;
	const hipsPositionScale = vrmHipsY / motionHipsHeight;

	clip.tracks.forEach((track) => {
		const trackSplitted = track.name.split(".");
		const mixamoRigName = trackSplitted[0];
		const vrmBoneName = mixamoVRMRigMap[mixamoRigName];
		const vrmNodeName = vrm.humanoid?.getNormalizedBoneNode(vrmBoneName)?.name;
		const mixamoRigNode = asset.getObjectByName(mixamoRigName);

		if (vrmNodeName != null && mixamoRigNode != null) {
			const propertyName = trackSplitted[1];

			mixamoRigNode.getWorldQuaternion(restRotationInverse).invert();
			mixamoRigNode.parent?.getWorldQuaternion(parentRestWorldRotation);

			if (track instanceof THREE.QuaternionKeyframeTrack) {
				for (let i = 0; i < track.values.length; i += 4) {
					const flatQuaternion = track.values.slice(i, i + 4);

					_quatA.fromArray(flatQuaternion);
					_quatA
						.premultiply(parentRestWorldRotation)
						.multiply(restRotationInverse);
					_quatA.toArray(flatQuaternion);

					flatQuaternion.forEach((v, index) => {
						track.values[index + i] = v;
					});
				}

				tracks.push(
					new THREE.QuaternionKeyframeTrack(
						`${vrmNodeName}.${propertyName}`,
						track.times,
						track.values.map((v, i) =>
							vrm.meta?.metaVersion === "0" && i % 2 === 0 ? -v : v,
						),
					),
				);
			} else if (track instanceof THREE.VectorKeyframeTrack) {
				const value = track.values.map(
					(v, i) =>
						(vrm.meta?.metaVersion === "0" && i % 3 !== 1 ? -v : v) *
						hipsPositionScale,
				);
				tracks.push(
					new THREE.VectorKeyframeTrack(
						`${vrmNodeName}.${propertyName}`,
						track.times,
						value,
					),
				);
			}
		}
	});

	return new THREE.AnimationClip("vrmAnimation", clip.duration, tracks);
}

### Documentation for `loadMixamoAnimation`

**Purpose and Behavior:**
The function `loadMixamoAnimation` loads an animation clip from a Mixamo FBX file, converts it to a format compatible with the VRM specification, and returns a THREE.js AnimationClip. It processes each track in the loaded animation, transforming quaternion and vector keyframes according to VRM bone mappings and scaling.

**Key Implementation Details:**
- **FBXLoader:** Loads the FBX asset from the provided URL.
- **Quaternion Transformations:** Adjusts quaternions based on parent and rest rotations for accurate animations.
- **Track Processing:** Handles both `QuaternionKeyframeTrack` and `VectorKeyframeTrack`, applying necessary transformations and scaling.

**When/Why to Use:**
Use this function when you need to import and adapt Mixamo animations into a VRM-compatible format. It ensures that the imported animations align with the VRM bone structure, making it suitable for integrating external animations into VRM models.

**Patterns/Gotchas:**
- The function relies on `mixamoVRMRigMap` for mapping Mixamo rig names to VRM bone names.
- Quaternion transformations are complex and critical; incorrect handling can lead to animation artifacts.
- Ensure the VRM model has a consistent meta version (`vrm.meta?.metaVersion`) to correctly apply scaling.
