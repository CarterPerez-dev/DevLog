def follow_up_abandoned_checkouts():
    resend.api_key = os.getenv("RESEND_API_KEY")
    if not resend.api_key:
        logger.error("RESEND_API_KEY not set, skipping follow-up task")
        return "Error: RESEND_API_KEY not set"

    cutoff = datetime.now(UTC) - timedelta(minutes=1)
    max_age = datetime.now(UTC) - timedelta(days=3)

    applications = MentorshipApplication.objects(
        status="accepted",
        applied_at__lt=cutoff,
        applied_at__gt=max_age,
    )

    if not applications:
        return "No abandoned checkouts to follow up"

    sent = 0
    errors = 0

    for app_doc in applications:
        try:
            first_name = (app_doc.name or "").split()[0] if app_doc.name else "there"

            prompt = ABANDONED_CHECKOUT_USER.format(
                name=first_name,
                current_situation=app_doc.current_situation or "",
                target_role=app_doc.target_role or "",
                certs=app_doc.certs or "None listed",
                obstacle=app_doc.obstacle or "",
                motivation=app_doc.motivation or "",
            )

            email_body = generate_text(
                prompt,
                system_prompt=ABANDONED_CHECKOUT_SYSTEM,
                max_completion_tokens=4000,
            )

            if not email_body:
                logger.warning("Empty AI response for application %s", str(app_doc.id))
                errors += 1
                continue

            html = _build_follow_up_html(first_name, email_body)

            resend.Emails.send({
                "from": f"Carter @ CertGames <{Email.MENTORSHIP_SENDER}>",
                "to": [app_doc.email],
                "subject": f"Hey {first_name} — quick question",
                "html": html,
                "reply_to": "support@certgames.com",
            })

            MentorshipApplication.objects(id=app_doc.id).update_one(
                set__status="followed_up"
            )

            sent += 1
            logger.info("Follow-up sent to %s (%s)", app_doc.email, str(app_doc.id))

        except Exception as e:
            logger.error("Failed follow-up for %s: %s", str(app_doc.id), str(e))
            errors += 1

    return f"Follow-up complete: {sent} sent, {errors} errors"
