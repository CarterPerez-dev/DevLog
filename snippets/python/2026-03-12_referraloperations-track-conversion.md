```python
def track_conversion(
        code: str,
        referred_user_id: str,
        subscription_id: str | None = None,
        amount: float | None = None,
        reward_override: float | None = None
    ) -> dict[str, Any]:
        with referral_db_session() as db:
            try:
                result = ReferralService.track_referral(
                    db = db,
                    code = code,
                    referred_user_id = str(referred_user_id),
                    transaction_id = subscription_id,
                    transaction_amount = amount
                )

                if reward_override is not None:
                    from stripe_referral.models.ReferralTracking import ReferralTracking  # pylint: disable=import-outside-toplevel
                    tracking = db.get(ReferralTracking, result["tracking_id"])
                    if tracking:
                        tracking.amount_earned = reward_override
                        result["amount_earned"] = reward_override

                current_app.logger.info(
                    f"Tracked referral: {code} -> user {referred_user_id}, "
                    f"earned ${result['amount_earned']}"
                )

            except SelfReferralError:
                current_app.logger.warning(
                    f"Self-referral attempted: {referred_user_id}"
                )
                raise
            except Exception as e:
                current_app.logger.error(
                    f"Failed to track referral for code {code}: {str(e)}"
                )
                raise

        try:
            payout_result = PayoutOperations.create_payouts_for_pending_trackings()
            current_app.logger.info(
                f"Created {payout_result.get('created', 0)} payout records"
            )
        except Exception as e:
            current_app.logger.error(
                f"Failed to create payout records: {str(e)}"
            )

        return result
```
