def build_notification_data(
        conversion_result: dict[str, Any],
        referred_user: Any,
        plan_type: str,
        plan_amount: float,
        reward_amount: float
    ) -> dict[str, Any]:
        referrer_user_id = conversion_result.get("referrer_user_id")

        referrer = User.objects(id=referrer_user_id).first() if referrer_user_id else None

        code_data = ReferralOperations.get_user_code(referrer_user_id) if referrer_user_id else None

        code_created_at = None
        if code_data and code_data.get("created_at"):
            with contextlib.suppress(ValueError, TypeError):
                code_created_at = datetime.fromisoformat(code_data["created_at"])

        return {
            "plan_type": plan_type,
            "plan_amount": plan_amount,
            "reward_amount": reward_amount,
            "referrer_username": referrer.username if referrer else "Unknown",
            "referrer_email": referrer.email if referrer else "Unknown",
            "referrer_account_created": referrer.createdAt if referrer else None,
            "referrer_subscription_active": referrer.subscriptionActive if referrer else False,
            "referrer_subscription_type": referrer.subscriptionType if referrer else None,
            "referrer_payment_method": referrer.paymentMethod if referrer else None,
            "referrer_total_referrals": code_data.get("uses_count", 0) if code_data else 0,
            "code_created_at": code_created_at,
            "referred_username": referred_user.username if referred_user else "Unknown",
            "referred_email": referred_user.email if referred_user else "Unknown",
            "referred_account_created": referred_user.createdAt if referred_user else None,
        }
