def update_ai_usage_log(
    log_id: str | None,
    *,
    input_tokens: int | None = None,
    output_tokens: int | None = None,
    processing_time: float | None = None,
    error_message: str | None = None,
    response_size: int | None = None,
) -> None:
    """
    Update an existing AI usage log with actual metrics after completion.
    """
    if not log_id:
        return

    try:
        log_entry = AIUsageLog.objects(id = log_id).first()
        if not log_entry:
            logger.warning("AI usage log not found: %s", log_id)
            return

        if input_tokens is not None:
            log_entry.input_tokens = input_tokens
        if output_tokens is not None:
            log_entry.output_tokens = output_tokens
        if input_tokens is not None and output_tokens is not None:

            log_entry.total_tokens = input_tokens + output_tokens
            log_entry.estimated_cost = calculate_ai_cost(
                log_entry.model or "gpt-5-nano",
                input_tokens,
                output_tokens
            )
        if processing_time is not None:
            log_entry.processing_time = processing_time
        if response_size is not None:
            log_entry.response_size = response_size

        if error_message:
            log_entry.error_message = error_message

        log_entry.save()

    except Exception as e:
        logger.error("Error updating AI usage log: %s", str(e))
