```go
func FetchStats() tea.Msg {
	var stats StatsMsg

	cpu, err := system.GetCPUUsage()
	if err != nil {
		stats.Err = err
		return stats
	}
	stats.CPU = cpu

	mem, err := system.GetMemoryInfo()
	if err != nil {
		stats.Err = err
		return stats
	}
	stats.Memory = mem

	disk, err := system.GetDiskUsage("/")
	if err != nil {
		stats.Err = err
		return stats
	}
	stats.Disk = disk

	gpuInfo, err := system.GetGPUInfo()
	if err == nil {
		stats.GPU = gpuInfo
	}

	gpuProcs, err := system.GetGPUProcesses()
	if err == nil {
		stats.GPUProcs = gpuProcs
	}

	procs, err := system.GetProcesses()
	if err != nil {
		stats.Err = err
		return stats
	}
	stats.Procs = procs

	return stats
}
```
