def repo(self) -> Repo:
        """
        Get or initialize the git repository
        """
        if self._repo is None:
            if not self.repo_path.exists():
                self.repo_path.mkdir(parents = True, exist_ok = True)

                self._repo = Repo(self.repo_path)
