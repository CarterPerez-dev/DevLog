def send_sync(self, content: str | Message) -> None:
        try:
            loop = asyncio.get_running_loop()
        except RuntimeError:
            loop = None

        if loop and loop.is_running():
            import concurrent.futures

            with concurrent.futures.ThreadPoolExecutor(max_workers = 1) as pool:
                pool.submit(asyncio.run, self.send(content)).result()
        else:
            asyncio.run(self.send(content))

By checking if an event loop is running, it either submits the task to a thread pool or directly runs it with `asyncio.run`, ensuring smooth execution regardless of the environment.

```python
def send_sync(self, content: str | Message) -> None:
    try:
        loop = asyncio.get_running_loop()
    except RuntimeError:
        loop = None

    if loop and loop.is_running():
        with concurrent.futures.ThreadPoolExecutor(max_workers=1) as pool:
            pool.submit(asyncio.run, self.send(content)).result()
    else:
        asyncio.run(self.send(content))
```
