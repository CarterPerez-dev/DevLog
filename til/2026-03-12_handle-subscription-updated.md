def handle_subscription_updated(subscription_data):
    """
    Handle subscription update events (webhook handler)
    """
    customer_id = subscription_data.get("customer")
    user_obj = SubscriptionService.find_by_stripe_customer_id(customer_id)
    user = user_obj.to_dict() if user_obj else None

    if not user:
        current_app.logger.warning(
            f"User not found for customer: {customer_id}"
        )
        return

    user_id = str(user["_id"])
    status = subscription_data.get("status")
    cancel_at_period_end = subscription_data.get(
        "cancel_at_period_end",
        False
    )
    canceled_at = subscription_data.get("canceled_at")

    status_mapping = {
        "active": "active",
        "past_due": "past_due",
        "canceled": "canceled",
        "unpaid": "unpaid",
        "incomplete": "incomplete",
        "incomplete_expired": "expired",
        "trialing": "trialing",
    }

    internal_status = status_mapping.get(status, status)
    is_active = internal_status in ["active", "trialing"]

    if is_active and cancel_at_period_end:
        internal_status = "canceling"

    end_timestamp = subscription_data.get(
        "cancel_at"
    ) or subscription_data.get("current_period_end")
    subscription_end_date = datetime.fromtimestamp(
        end_timestamp
    ) if end_timestamp else None

    update_data = {
        "subscriptionActive": is_active,
        "subscriptionStatus": internal_status,
        "subscriptionPlatform": "stripe",
        "stripeSubscriptionId": subscription_data.get("id"),
        "subscriptionEndDate": subscription_end_date,
    }

    if cancel_at_period_end and canceled_at:
        update_data["subscriptionCanceledAt"] = datetime.fromtimestamp(
            canceled_at
        )
    elif is_active and not cancel_at_period_end:
        update_data["subscriptionCanceledAt"] = None

    SubscriptionService.update_subscription_data(user_id, update_data)

    AuditLogger.log_action(
        "subscription_updated",
        user_id = user_id,
        data = {
            "platform": "stripe",
            "subscription_id": subscription_data.get("id"),
            "status": status,
            "internal_status": internal_status,
        },
    )
