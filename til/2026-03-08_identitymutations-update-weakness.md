```python
async def update_weakness(
        session: AsyncSession,
        weakness_id: UUID,
        data: dict
    ) -> IdentityWeakness:
        """
        Update weakness - partial update supported
        """
        result = await session.execute(
            select(IdentityWeakness).where(
                IdentityWeakness.id == weakness_id
            )
        )
        weakness = result.scalar_one()

        if "weakness" in data:
            weakness.weakness = data["weakness"]
        if "impact" in data:
            weakness.impact = data["impact"]
        if "workaround" in data:
            weakness.workaround = data["workaround"]

        await session.flush()
        await session.refresh(weakness)
        return weakness
```

```python
async def update_weakness(
        session: AsyncSession,
        weakness_id: UUID,
        data: dict
    ) -> IdentityWeakness:
        result = await session.execute(
            select(IdentityWeakness).where(
                IdentityWeakness.id == weakness_id
            )
        )
        weakness = result.scalar_one()
        
        for key in ["weakness", "impact", "workaround"]:
            if key in data:
                setattr(weakness, key, data[key])
                
        await session.flush()
        await session.refresh(weakness)
        return weakness
```

This approach reduces redundancy and enhances readability.
