func (d Dashboard) Update(msg tea.Msg) (Dashboard, tea.Cmd) {
	switch msg := msg.(type) {
	case StatsMsg:
		d.cpu = msg.CPU
		d.memory = msg.Memory
		d.disk = msg.Disk
		d.gpu = msg.GPU
		d.gpuProcs = msg.GPUProcs
		d.procs = msg.Procs
		d.err = msg.Err
	case tea.KeyMsg:
		switch msg.String() {
		case "m":
			d.sortMode = SortByMemory
		case "c":
			d.sortMode = SortByCPU
		case "g":
			d.sortMode = SortByGPU
		case "n":
			d.sortMode = SortByName
		}
	}
	return d, nil
}
