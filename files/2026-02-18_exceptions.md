        self,

        status_code: int = 500,
        extra: dict[str,
                    Any] | None = None,
    ) -> None:

class ResourceNotFound(BaseAppException):
    """
    Raised when a requested resource does not exist
    """

        self,
        resource: str,
        identifier: str | int,
        extra: dict[str,
                    Any] | None = None,

            extra = extra,
