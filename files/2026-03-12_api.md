    doc = '/docs/',
    authorizations = {
        'BearerAuth': {
            'type': 'apiKey',
            'in': 'header',
            'name': 'Authorization',
            'description': 'Enter your JWT in the format: Bearer <token>'
        },
        'CSRFToken': {
            'type': 'apiKey',
            'in': 'header',
            'name': 'X-Cracked',
            'description': 'CSRF token for admin endpoints'

    contact = 'support@certgames.com',
