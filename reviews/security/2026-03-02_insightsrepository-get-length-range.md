def get_length_range_label(seconds: int) -> str:
        """Get human-readable length range label"""
        ranges = [
            (30,
             "0:00-0:30"),
            (60,
             "0:30-1:00"),
            (90,
             "1:00-1:30"),
            (120,
             "1:30-2:00"),
            (150,
             "2:00-2:30"),
            (180,
             "2:30-3:00"),
        ]
        for max_sec, label in ranges:
            if seconds <= max_sec:
                return label
        return "3:00+"
