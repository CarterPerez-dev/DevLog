func ParseFile(path string) ([]types.Dependency, error) {
	f, err := os.Open(path) //nolint:gosec
	if err != nil {
		if os.IsNotExist(err) {
			return nil, fmt.Errorf(
				"open %s: file not found", path,
			)
		}
		return nil, fmt.Errorf("read %s: %w", path, err)
	}
	defer func() { _ = f.Close() }() //nolint:errcheck

	var deps []types.Dependency
	scanner := bufio.NewScanner(f)

	for scanner.Scan() {
		line := strings.TrimSpace(scanner.Text())

		if line == "" ||
			strings.HasPrefix(line, "#") ||
			strings.HasPrefix(line, "-") {
			continue
		}

		if idx := strings.Index(line, " #"); idx >= 0 {
			line = strings.TrimSpace(line[:idx])
		}

		dep := parseLine(line)
		if dep.Name != "" {
			deps = append(deps, dep)
		}
	}

	if err := scanner.Err(); err != nil {
		return nil, fmt.Errorf("read %s: %w", path, err)
	}

	if len(deps) == 0 {
		return nil, fmt.Errorf(
			"%s: no dependencies found", path,
		)
	}
	return deps, nil
}

## Performance Analysis

### Performance Analysis

**Time Complexity**: The function has a time complexity of \(O(n)\), where \(n\) is the number of lines in the file, due to the single pass through each line.

**Space Complexity**: The space complexity is also \(O(n)\) for storing the dependencies. However, the `deps` slice can grow significantly if there are many valid dependency entries.

**Bottlenecks and Inefficiencies**:
- **Error Handling**: The error handling could be improved to return more specific errors.
- **Redundant Operations**: The `strings.TrimSpace()` call is redundant as `scanner.Text()` already returns a trimmed string. 
- **Unnecessary Iterations**: The check for empty lines, comments, and prefix checks can be combined into one condition.
