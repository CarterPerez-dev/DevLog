```tsx
function UserStats(): JSX.Element {
  const { user } = useAuthStore();
  const { isSimple } = useTheme();
  const s = isSimple ? simpleStyles : defaultStyles;
  const { data: levelProgress } = useGetLevelProgress();

  if (user === null || user === undefined) {
    return <div className={s.userStats} />;
  }

  const userXP = user.xp ?? 0;
  const userLevel = user.level ?? 1;
  const userCoins = user.coins ?? 0;

  const xpPercentage = levelProgress?.progressPercentage ?? 0;

  return (
    <div className={s.userStats}>
      <div className={s.statItem}>
        <GiDualityMask className={s.icon} />
        <div className={s.content}>
          <span className={s.label}>LVL</span>
          <span className={s.value}>{userLevel}</span>
        </div>
      </div>

      <div className={s.statItem}>
        <GiHawkEmblem className={s.icon} />
        <div className={s.content}>
          <span className={s.label}>XP</span>
          <span className={s.value}>{formatLargeNumber(userXP)}</span>
        </div>
        <div className={s.xpBar}>
          <div
            className={s.xpFill}
            style={{ width: `${xpPercentage.toString()}%` }}
          />
        </div>
      </div>

      <div className={s.statItem}>
        <GiOuroboros className={s.icon} />
        <div className={s.content}>
          <span className={s.label}>COINS</span>
          <span className={s.value}>
            {formatLargeNumber(userCoins)}
          </span>
        </div>
      </div>
    </div>
  );
}
```
