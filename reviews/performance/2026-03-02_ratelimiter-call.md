```python
async def __call__(self, message: Message, next_: SendFunc) -> None:
        self._refill()

        if self._tokens >= 1.0:
            self._tokens -= 1.0
            if self._dropped_count > 0:
                message.metadata["dropped_count"] = self._dropped_count
                self._dropped_count = 0
            await next_(message)
        else:
            self._dropped_count += 1
```
